function saveData() { const BOT_USERNAME = "ТВОЙ_ЮЗЕРНЕЙМ_БОТА"; // Например: RavenGuardBot (без @) let nick = document.getElementById("nick").value.trim(); let res = document.getElementById("res").value; if (!nick || !res) { window.Telegram.WebApp.showAlert("Заполните ник и стойкость!"); return; } // Собираем данные в объект let payload = { n: nick, r: parseFloat(res), f: parseFloat(document.getElementById("fc").value || 0), s: parseFloat(document.getElementById("sc").value || 0) }; // Кодируем данные в Base64, чтобы Telegram пропустил их в ссылке let encodedData = btoa(JSON.stringify(payload)); // Формируем ссылку для перехода let link = "https://t.me/" + BOT_USERNAME + "?start=" + encodedData; // Выполняем переход и закрываем окно window.Telegram.WebApp.openTelegramLink(link); window.Telegram.WebApp.close(); }
